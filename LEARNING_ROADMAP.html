class BankAccount:
    def __init__(self, owner, balance=0):
        """初始化账户，owner=户主名，balance=余额"""
        self.owner = owner
        self.balance = balance

    def deposit(self, amount):
        """存钱"""
        if amount > 0:
            self.balance += amount
            print(f"存了 {amount}，余额 {self.balance}")
        else:
            print("存款必须是正数")

    def withdraw(self, amount):
        """取钱"""
        if amount > self.balance:
            print(f"余额不足！余额 {self.balance}")
        else:
            self.balance -= amount
            print(f"取了 {amount}，余额 {self.balance}")

    def __str__(self):
        """print(账户) 时显示的内容"""
        return f"账户：{self.owner}，余额：{self.balance}"

def safe_divide(a, b):
    """安全除法，除数为0时返回 None 而不是崩溃"""
    try:
        result = a / b
        return result
    except ZeroDivisionError:
        print("错误：除数不能为0")
        return None
    except TypeError:
        print("错误：类型不对，不能除")
        return None

print(safe_divide(10, 2))   # 5.0
print(safe_divide(10, 0))    # 打印错误，返回 None
print(safe_divide("a", 2))  # 打印错误，返回 None

from langgraph.graph import StateGraph, END
from typing import TypedDict

class MyState(TypedDict):
    message: str
    step_count: int

def node_a(state: MyState) -> MyState:
    state["step_count"] += 1
    state["message"] = state["message"] + " → A"
    return state

def node_b(state: MyState) -> MyState:
    state["step_count"] += 1
    state["message"] = state["message"] + " → B"
    return state

def should_continue(state: MyState) -> str:
    """根据 step_count 决定下一步"""
    if state["step_count"] < 2:
        return "node_a"
    return END

# 创建图
graph = StateGraph(MyState)
graph.add_node("node_a", node_a)
graph.add_node("node_b", node_b)
graph.set_entry_point("node_a")
graph.add_conditional_edges("node_a", should_continue, {"node_a": "node_a", END: END})
graph.add_edge("node_b", END)

# 运行
app = graph.compile()
result = app.invoke({"message": "开始", "step_count": 0})
print("最终状态:", result)
print("经过的节点:", result["message"])

def call_api_may_fail():
    """模拟可能失败的 API 调用，成功率 40%"""
    if random.random() < 0.4:
        return {"success": True, "data": "查询结果：各部门使用时长"}

    return None

from rank_bm25 import BM25Okapi
import os

class UnifiedRetriever:
    def __init__(self):
        # ... 原有的 Milvus 初始化 ...
        # 新增：BM25 索引
        self._bm25: Optional[BM25Okapi] = None
        self._bm25_texts: List[str] = []

    def _ensure_bm25(self):
        """构建 BM25 索引（只需构建一次）"""
        if self._bm25 is not None:
            return
        # 从 Milvus 读取所有 Schema 文本
        all_data = self._client.query(
            collection_name=settings.milvus_collection_schema,
            output_fields=["text"],
            limit=1000,
        )
        self._bm25_texts = [d["text"] for d in all_data]
        tokenized = [t.split() for t in self._bm25_texts]
        self._bm25 = BM25Okapi(tokenized)

    def search(self, query: str, top_k: int = 5):
        """两路检索 + RRF 融合"""
        # 1. 向量检索（已有）
        vector_results = self._vector_search(query, top_k * 2)

        # 2. BM25 检索（新增）
        self._ensure_bm25()
        tokenized_query = query.split()
        bm25_scores = self._bm25.get_scores(tokenized_query)
        top_bm25_idx = sorted(range(len(bm25_scores)), key=lambda i: bm25_scores[i], reverse=True)[:top_k * 2]

        # 3. RRF 融合（Reciprocal Rank Fusion）
        rrf_scores = {}
        for rank, (res,) in enumerate(vector_results):
            rrf_scores[res["id"]] = rrf_scores.get(res["id"], 0) + 1 / (60 + rank)
        for rank, idx in enumerate(top_bm25_idx):
            bm25_id = self._bm25_texts[idx]
            rrf_scores[bm25_id] = rrf_scores.get(bm25_id, 0) + 1 / (60 + rank)

        # 取 top_k
        final_top = sorted(rrf_scores.items(), key=lambda x: x[1], reverse=True)[:top_k]
        return final_top
